    ```go
    send.View(c, view.View{
       Name: "Welcome",
       RenderMode: view.RenderModeFull,
       // RenderMode: view.RenderModeServer,
       // RenderMode: view.RenderModeClient,
    })
    ```
    <LinkCard title='Read More' href="/frizzante-docs/guides/views/#render-modes"/>
  </Card>
  <Card title='Web Sockets' icon='heart'>

    ```go
    send.WsUpgrade(c)
    for {
       n := receive.Message(c)
       send.Message(c, "Hello " + n)
    }
    ```
    <LinkCard title='Read More' href="/frizzante-docs/guides/web-sockets"/>
  </Card>
  <Card title='Server Sent Events' icon='heart'>

    ```go

    for {
       send.Message(c, "Hello.")
       time.Sleep(time.Second)
    }
    ```
    <LinkCard title='Read More' href="/frizzante-docs/guides/web-sockets"/>
